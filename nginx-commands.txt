# SSH into your Ubuntu server first, then run these commands:

# 1. Create/update the Nginx config
sudo bash -c 'cat > /etc/nginx/sites-available/trading-os << NGINX
server {
    listen 80;
    server_name api.roshanvijay.com;

    location / {
        proxy_pass http://localhost:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade \$http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
    }

    location /ws/ {
        proxy_pass http://localhost:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade \$http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host \$host;
        proxy_read_timeout 86400;
    }
}
NGINX'

# 2. Enable the site
sudo ln -sf /etc/nginx/sites-available/trading-os /etc/nginx/sites-enabled/

# 3. Test config and reload
sudo nginx -t && sudo systemctl reload nginx

# 4. Done!
echo "Nginx updated for WebSocket support"
